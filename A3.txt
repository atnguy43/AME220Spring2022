34 = 3*16 + 4 = 52

2) rbg -> Hex


rbg(213, 12,112) = #D50C70

213	= 13*(16^1)  + 5*(16^0) = #D5
12	= 0*16 + 12*1 = #0C
112	= 7*16 + 0*1 = #70


rgb(22,126,52) = #167E34

22	= 1*16 + 6*1 = #16
126	= 7*16 + 14*1 = #7E
52	= 3*16 + 4*1 = #34


rgb(213,214,41) = #D5D629

213	= 13*16 + 5*1 = #D5
214	= 13*16 + 6*1 = #D6
41	= 2*16 + 9*1 = #29


rgb(113,22,113) = #711671

113	= 7*16 + 1*1 = #71
22	= 1*16 + 6*1 = #16
113	= 7*16 + 1*1 = #71


rgb(126,112,172) = #7E70AC

126	= 7*16 + 14*1 = #7E
112	= 7*16 + 0*1 = #70
172	= 10*16 + 12*1 = #AC


rgb(142,24,241) = #8E18F1

142	= 8*16 + 14*1 = #8E
24	= 1*16 + 8*1 = #18
241	= 15*16 + 1*1 = #F1


rgb(123,32,121) = #7B2079

123	= 7*16 + 11*1 = #7B
32	= 2*16 + 0*1 = #20
121	= 7*16 + 9*1 = #79


rgba(127,123,224,.4) =#7F7BE066

127	= 7*16 + 15*1 = #7F
123	= 7*16 + 11*1 = #7B
224	= 14*16 + 0*1 = #E0
.4	= (.4*100)*(255/100) = 102 --- To hex ---> floor(102/16) + (102 mod 16) = #66
